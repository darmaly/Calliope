    ```typescript
    function collectMidiEvents(timelineState): MidiEvent[] {
      const events: MidiEvent[] = []
      for (const track of timelineState.tracks) {
        for (const clip of Object.values(timelineState.clips).filter(c => c.trackId === track.id)) {
          if (clip.type === 'midi' && clip.notes) {
            for (const note of clip.notes) {
              events.push({
                beatPosition: clip.startBeat + note.startBeat,
                noteNumber: note.pitch,
                velocity: note.velocity,
                durationBeats: note.durationBeats,
                trackId: track.id
              })
            }
          }
        }
      }
      return events
    }
    ```
