    ```cpp
    #pragma once
    #include <juce_audio_formats/juce_audio_formats.h>
    #include <juce_audio_processors/juce_audio_processors.h>
    #include <functional>
    #include <string>

    namespace calliope {
    class Engine;

    enum class ExportFormat { WAV_16, WAV_24, MP3, FLAC };

    struct ExportOptions {
        std::string outputPath;
        ExportFormat format = ExportFormat::WAV_16;
        int mp3Bitrate = 192;         // kbps (128, 192, 256, 320)
        double totalBeats = 0.0;      // length in beats (0 = auto-detect from project)
        double sampleRate = 0.0;      // 0 = use project rate
    };

    struct StemExportOptions {
        std::string outputDir;        // directory for stems
        double totalBeats = 0.0;
        double sampleRate = 0.0;
    };

    using ProgressCallback = std::function<void(float)>; // 0.0 to 1.0

    class AudioExporter {
    public:
        explicit AudioExporter(Engine& engine);

        // Export full mix to file. Returns true on success.
        bool exportMix(const ExportOptions& options, ProgressCallback progress = nullptr);

        // Export individual track stems as WAV. Returns true on success.
        bool exportStems(const StemExportOptions& options, ProgressCallback progress = nullptr);

    private:
        Engine& engine_;

        // Render the graph offline into an AudioBuffer
        juce::AudioBuffer<float> offlineBounce(double totalBeats, double sampleRate,
                                                int bufferSize, ProgressCallback progress);

        // Render a single track by soloing it (muting others) — NOTE: use isolated rendering approach
        juce::AudioBuffer<float> offlineBounceSingleTrack(const juce::String& trackId,
                                                           double totalBeats, double sampleRate,
                                                           int bufferSize);

        bool writeWav(const juce::File& file, const juce::AudioBuffer<float>& buffer,
                      double sampleRate, int bitDepth);
        bool writeFlac(const juce::File& file, const juce::AudioBuffer<float>& buffer,
                       double sampleRate);
        bool writeMp3(const juce::File& file, const juce::AudioBuffer<float>& buffer,
                      double sampleRate, int bitrate);
    };
    } // namespace calliope
    ```

    Create `engine/src/audio_exporter.cpp`:
    - `offlineBounce()`: Stop transport. Set transport to position 0. Prepare graph for offline rendering by calling `prepareToPlay(sampleRate, bufferSize)` on the graph. Calculate totalSamples from totalBeats and BPM and sampleRate. Loop: clear buffer, advance transport manually (incrementing ppqPosition by bufferSize samples worth of beats), call `graph_.processBlock(buffer, midi)`, write samples. Report progress via callback. Per Research Pitfall 3: ensure graph connections are fully initialized before starting the loop.
    - `writeWav()`: Use `juce::WavAudioFormat`, create writer via `createWriterFor(new juce::FileOutputStream(file), sampleRate, channels, bitDepth, {}, 0)`, write with `writeFromAudioSampleBuffer()`.
    - `writeFlac()`: Use `juce::FlacAudioFormat`, same pattern as WAV. FLAC is always 16-bit or 24-bit integer; use 16-bit default.
    - `writeMp3()`: Include `<lame/lame.h>`. Guard with `#ifdef CALLIOPE_HAS_LAME`. Init LAME with `lame_init()`, set sample rate, channels, bitrate, quality=2. Read from the AudioBuffer in chunks, encode with `lame_encode_buffer_interleaved_ieee_float()`, write to FileOutputStream, flush with `lame_encode_flush()`, close with `lame_close()`.
    - `exportMix()`: Call offlineBounce(), then write to format based on ExportOptions.
    - `exportStems()`: For each track (polySynth, bassSynth, drumMachine), render individually by muting all others in the offline bounce, then write as WAV to `{outputDir}/{trackName}.wav`. Restore state after. Per user decision: stems/ subfolder.
    - `offlineBounceSingleTrack()`: Temporarily solo the target track. This is OFFLINE only — no audio device involved. After rendering, un-solo.

    IMPORTANT for offline bounce: The current AudioGraph is connected to an audio device. For offline rendering, either (a) stop playback and disconnect from device, or (b) create a temporary rendering context. Use approach (a) per Research recommendation: stop transport, use graph.processBlock directly in a loop. The AudioProcessorGraph can be called from any thread when not connected to an audio callback.

    IMPORTANT MIDI scheduling: Since MIDI notes live in Zustand (renderer), the export function must accept MIDI event data as a parameter. Add `std::vector<MidiEventData>` to ExportOptions where MidiEventData = { double beatPosition, int noteNumber, int velocity, double durationBeats, string trackId }. The offlineBounce loop schedules these into MidiBuffer for each processBlock call based on current beat position.
