    ```cpp
    } else if (command == "effect.insert") {
        auto trackId = juce::String(params.Get("trackId").As<Napi::String>().Utf8Value());
        auto effectType = juce::String(params.Get("effectType").As<Napi::String>().Utf8Value());
        int position = params.Has("position") ? params.Get("position").As<Napi::Number>().Int32Value() : -1;
        auto& chain = engine.getInsertChain(trackId);
        cmd = std::make_unique<calliope::InsertEffectCommand>(chain, engine, trackId, effectType, position);
    } else if (command == "effect.remove") {
        auto trackId = juce::String(params.Get("trackId").As<Napi::String>().Utf8Value());
        int position = params.Get("position").As<Napi::Number>().Int32Value();
        auto& chain = engine.getInsertChain(trackId);
        cmd = std::make_unique<calliope::RemoveEffectCommand>(chain, engine, trackId, position);
    } else if (command == "effect.reorder") {
        auto trackId = juce::String(params.Get("trackId").As<Napi::String>().Utf8Value());
        int fromPosition = params.Get("fromPosition").As<Napi::Number>().Int32Value();
        int toPosition = params.Get("toPosition").As<Napi::Number>().Int32Value();
        auto& chain = engine.getInsertChain(trackId);
        cmd = std::make_unique<calliope::ReorderEffectCommand>(chain, engine, trackId, fromPosition, toPosition);
    } else if (command == "effect.bypass") {
        auto trackId = juce::String(params.Get("trackId").As<Napi::String>().Utf8Value());
        int position = params.Get("position").As<Napi::Number>().Int32Value();
        bool bypassed = params.Get("bypassed").As<Napi::Boolean>().Value();
        auto& chain = engine.getInsertChain(trackId);
        cmd = std::make_unique<calliope::BypassEffectCommand>(chain, position, bypassed);
    }
    ```
