    **use-meter-polling.ts** — requestAnimationFrame hook
    ```typescript
    import { useEffect } from 'react'
    import { useMixerStore } from '../stores/mixer-store'

    export function useMeterPolling() {
      const panelVisible = useMixerStore((s) => s.panelVisible)
      const updateMeterLevels = useMixerStore((s) => s.updateMeterLevels)

      useEffect(() => {
        if (!panelVisible) return
        let rafId: number
        const poll = async () => {
          try {
            const levels = await window.calliope.getMeterLevels()
            updateMeterLevels(levels)
          } catch { /* engine not ready */ }
          rafId = requestAnimationFrame(poll)
        }
        rafId = requestAnimationFrame(poll)
        return () => cancelAnimationFrame(rafId)
      }, [panelVisible, updateMeterLevels])
    }
    ```
